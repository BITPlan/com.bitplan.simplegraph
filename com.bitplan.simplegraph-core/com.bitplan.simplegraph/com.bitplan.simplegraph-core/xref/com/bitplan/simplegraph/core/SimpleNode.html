32  /**
33   * The SimpleNode interface represents a Vertex in a graph
34   * https://en.wikipedia.org/wiki/Vertex_(graph_theory) it wraps an Apache
35   * TinkerPop vertex
36   * 
37   * @author wf
38   *
39   */
40  public interface SimpleNode extends SimpleGraph {
41    
42    static String SELF_LABEL="mysimplenode";
43    
44    static Consumer<Vertex> printDebug = vertex -> vertex.properties()
45        .forEachRemaining(prop -> System.out.println(String.format("%s.%s=%s",
46            vertex.label(), prop.label(), prop.value())));
47    
48    static Consumer<Vertex> printObjectDebug = vertex -> vertex.properties()
49        .forEachRemaining(prop -> System.out.println(String.format("%s.%s=%s (%s)",
50            vertex.label(), prop.label(), prop.value(),prop.value().getClass().getName())));
51  
52    static Consumer<Map<?, Object>> printMapDebug = map -> {
53      map.keySet().forEach(
54          key -> System.out.println(String.format("%s=%s", key, map.get(key))));
55    };

82    /**
83     * set the property value for the given key with traditional java style syntax
84     * 
85     * @param key
86     * @param value
87     */
88    public default void setProperty(String key, Object value) {
89      property(key, value);
90    }

92    /**
93     * get the property value for the given key with traditional java style syntax
94     * 
95     * @param key
96     * @return - the property value
97     */
98    public default Object getProperty(String key) {
99      return property(key);
100   }

102   /**
103    * add the key value pair to the vertex and map tinkerpop style syntax
104    * 
105    * @param key
106    * @param value
107    */
108   public default void property(String key, Object value) {
109     getVertex().property(key, value);
110     getMap().put(key, value);
111   }

114    * tinkerpop compatible getter
115    * 
116    * @param key
117    * @return - the property
118    */
119   public default Object property(String key) {
120     return getMap().get(key);
121   }

123   // interfaces with default implementation
124   /**
125    * recursive out handling
126    * 
127    * @param edgeName
128    * @param recursionDepth
129    * @return
130    */
131   public default Stream<SimpleNode> recursiveOut(String edgeName,
132       int recursionDepth) {
133     // get the neighbor nodes with wrapped vertices following the edge with the
134     // given name
135     // prepare a list of simple nodes for the recursive results
136     List<SimpleNode> recursiveOuts = new ArrayList<SimpleNode>();
137     Stream<SimpleNode> outs = this.out(edgeName);
138     // if we still have recursion levels left over
139     if (recursionDepth > 0) {
140       outs.forEach(simpleNode -> {
141         recursiveOuts.add(simpleNode);
142         // get the edge nodes for this level
143         Stream<SimpleNode> levelOuts = simpleNode.recursiveOut(edgeName,
144             recursionDepth - 1);
145         // add them all to the recursive result
146         levelOuts
147             .forEach(levelSimpleNode -> recursiveOuts.add(levelSimpleNode));
148       });
149     }
150     return recursiveOuts.stream();
151   }
152 
153   // show name values
154   public default void printNameValues(PrintStream out) {
155     Map<String, Object> map = this.getMap();
156     for (String key : map.keySet()) {
157       out.println(String.format("%s = %s", key, map.get(key)));
158     }
159   }

161   /**
162    * forAll
163    */
164   public default void forAll(Consumer<Vertex> consumer) {
165     g().V().forEachRemaining(consumer);
166   };
167   
168   /**
169    * get the selfLabel for this Node
170    * @return - the self label
171    */
172   public default String getSelfLabel() {
173     return SELF_LABEL;
174   }
175   
176   /**
177    * get the Keys
178    * @return the keys
179    */
180   public Keys getKeys();
181 
182   /**
183    * get the SimpleNode of a Vertex 
184    * @param vertex
185    * @return - the simpleNode if available or null if not
186    */
187   public static SimpleNode of(Vertex vertex) {
188     if (vertex.property(SimpleNode.SELF_LABEL).isPresent()) {
189       return (SimpleNode) vertex.property(SimpleNode.SELF_LABEL).value();
190     }
191     return null;
192   }
