36  /**
37   * test the rythm template engine
38   * 
39   * @author wf
40   *
41   */
42  public class TestRythm  {
43    public static boolean debug = false;
44    protected static Logger LOGGER = Logger.getLogger("com.bitplan.simplegraph");
45    
46    @Test
47    public void testRythmFromFile() throws Exception {
48      RythmContext rythmContext = RythmContext.getInstance();
49      File templateDir = new File("../simplegraph-core/src/main/rythm");
50      rythmContext.setTemplateRoot(templateDir.getAbsolutePath());
51      // set it again - the engine should not be reconfigured
52      rythmContext.setTemplateRoot(templateDir.getAbsolutePath());
53      File template = new File(templateDir, "test.rythm");
54      Map<String, Object> rootMap = new HashMap<String, Object>();
55      rootMap.put("title", "testTitle");
56      String result = rythmContext.render(template, rootMap);
57      assertEquals(result.trim(), "testTitle");
58    }
59  
60    @Test
61    public void testRythmFromString() throws Exception {
62      // debug=true;
63      RythmContext rythmContext = RythmContext.getInstance();
64      String template = "@// Rythem template\n"
65          + "@// you can try me out at http://fiddle.rythmengine.com\n"
66          + "@// Created by Wolfgang Fahl, BITPlan GmbH,  2018-01-12\n"
67          + "@args() {\n" + "  String title,int times;\n" + "}\n"
68          + "@for (int i=0;i<times;i++) {\n" + "  @title @i\n" + "}";
69      if (debug) {
70        // if there is something wrong with the template try it out in the Rythm
71        // fiddle
72        LOGGER.log(Level.INFO, template);
73      }
74      Map<String, Object> rootMap = new HashMap<String, Object>();
75      rootMap.put("title", "step");
76      rootMap.put("times", 5);
77      String result = rythmContext.render(template, rootMap);
78      if (debug)
79        LOGGER.log(Level.INFO, result);
80      assertTrue(result.contains("step 4"));
81    }
82  }
