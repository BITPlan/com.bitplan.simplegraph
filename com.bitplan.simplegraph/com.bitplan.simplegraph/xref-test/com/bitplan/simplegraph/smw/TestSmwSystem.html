207   @Test
208   public void testDataTypeAsConcept() throws Exception {
209     // see
210     // https://www.semantic-mediawiki.org/w/index.php?title=User:WolfgangFahl/Workdocumentation_2018-01-02&action=edit
211     String askQuery = "{{#ask: [[Has_annotation_uri::+]]\n"
212         + "|mainlabel=page\n"
213         + "|?Has_annotation_uri=anu\n" + "|?Has_boolean=boo\n"
214         + "|?Has_code=cod\n" + "|?Has_date=dat\n" + "|?Has email address=ema\n"
215         + "|?Has Wikidata item ID=eid\n" + "|?Has coordinates=geo\n"
216         + "|?Has number=num\n" + "|?Has mlt=mlt\n" + "|?Has example=wpg\n"
217         + "|?Telephone number=tel\n" + "|?Has temperatureExample=tem\n"
218         + "|?Area=qty\n" + "|?SomeProperty=txt\n" + "|?Soccer result=rec\n"
219         + "|?Has_URL=uri\n" + "|format=ol\n" + "}}";
220     SmwSystem smwSystem = getSMWSystem();
221     // debug = true;
222     smwSystem.setDebug(debug);
223     JsonSystem askJson = smwSystem.getAskJsonResult(askQuery);
224     smwSystem.conceptAlizePrintRequests("datatype", askJson);
225     if (debug)
226       smwSystem.forAll(SimpleNode.printDebug);
227     assertEquals(2,
228         smwSystem.g().V().hasLabel("datatype").count().next().longValue());
229     if (debug)
230       smwSystem.g().V().hasLabel("datatype")
231           .forEachRemaining(SimpleNode.printDebug);
232     Vertex dtV = smwSystem.g().V().hasLabel("datatype").next();
233     assertEquals("datatype", dtV.property("isA").value().toString());
234     assertEquals("https://www.semantic-mediawiki.org",
235         dtV.property("anu").value().toString());
236     assertTrue(dtV.property("boo").value() instanceof Boolean);
237     assertTrue((Boolean) dtV.property("boo").value());
238     assertTrue(dtV.property("cod").value().toString().startsWith("Code"));
239     assertTrue(dtV.property("dat").value().toString()
240         .startsWith("Fri May 22 17:32:00"));
241     if (dtV.property("eid").isPresent())
242       assertEquals("Q9682", dtV.property("eid").value().toString());
243     assertEquals("mailto:president@whitehouse.gov",
244         dtV.property("ema").value().toString());
245     assertTrue(dtV.property("geo").value() instanceof Geo);
246     assertEquals("  32° 42’ 54.00” N  117°  9’ 45.00” W",
247         dtV.property("geo").value().toString());
248     assertTrue(dtV.property("page").value() instanceof WikiPage);
249     assertTrue(dtV.property("page").value().toString().startsWith("Help:"));
250     assertTrue(dtV.property("qty").value() instanceof Qty);
251     assertEquals(" 1052.00 km²",dtV.property("qty").value().toString());   
252     assertTrue(dtV.property("wpg").value() instanceof WikiPage);
253     assertEquals("Semantic MediaWiki", dtV.property("wpg").value().toString());
254     assertEquals("Did you create the page for Tokyo 東京 ? Yes ✓",
255         dtV.property("txt").value().toString());
256     assertEquals("http://www.whitehouse.gov",dtV.property("uri").value().toString());
257   }

259   @Test
260   public void testTimeFormat() throws ParseException {
261     long timestamps[] = { 1432315920, 1519171200 };
262     String timeraws[] = { "1/2015/5/22/17/32/0/0", "1/2018/2/21" };
263     String expected[] = { "2015-05-22 17:32:00", "2018-02-21 00:00:00" };
264     for (int i = 0; i < timeraws.length; i++) {
265       long timestamp = timestamps[i] * 1000;
266       Date timeRawDate = SmwSystem.getTime(timeraws[i]);
267       Date timeStampDate = new Date(timestamp);
268       SimpleDateFormat isoDate = new SimpleDateFormat("yyyy-MM-dd HH:mm:ss");
269       assertEquals(expected[i], isoDate.format(timeRawDate));
270       // assertEquals(timeRawDate,timeStampDate);
271       if (debug) {
272         System.out.println((timestamp - timeRawDate.getTime()));
273         System.out.println(isoDate.format(timeStampDate));
274         System.out.println(isoDate.format(timeRawDate));
275       }
276     }
277   }

297   @Test
298   public void testSimpleGraphModules() throws Exception {
299     String askQuery = "{{#ask: [[Concept:SimpleGraphModule]]\n"
300         + "|mainlabel=SimpleGraphModule\n"
301         + "| ?SimpleGraphModule name = name\n"
302         + "| ?SimpleGraphModule logo = logo\n"
303         + "| ?SimpleGraphModule modulename = modulename\n"
304         + "| ?SimpleGraphModule systemname = systemname\n"
305         + "| ?SimpleGraphModule url = url\n"
306         + "| ?SimpleGraphModule apiname = apiname\n"
307         + "| ?SimpleGraphModule apiurl = apiurl\n"
308         + "| ?SimpleGraphModule documentation = documentation\n" + "}}";
309     SmwSystem smw = new SmwSystem();
310     // debug = true;
311     smw.setDebug(debug);
312     smw.connect("http://wiki.bitplan.com", "/");
313     smw.moveTo("ask=" + askQuery);
314     if (debug)
315       smw.forAll(SimpleNode.printDebug);
316     List<Vertex> moduleVs = smw.getStartNode().g().V()
317         .hasLabel("SimpleGraphModule").toList();
318     RythmContext rythmContext = RythmContext.getInstance();
319     Map<String, Object> rootMap = new HashMap<String, Object>();
320     rootMap.put("moduleVs", moduleVs);
321     String template = "@import org.apache.tinkerpop.gremlin.structure.Vertex\n"
322         + "@import com.bitplan.simplegraph.smw.SmwSystem.WikiPage\n"
323         + "@args {\n" + "  List<Vertex> moduleVs\n" + "}\n"
324         + "@def WikiPage wikipage(Vertex v, String propname){\n"
325         + " return (WikiPage)v.property(propname).value();\n" + "}\n"
326         + "@def link(Vertex v, String propname) {\n@{ WikiPage wp=wikipage(v,propname)} URL=\"[[@(wp.fulltext)]]\"}\n"
327         + "@def image(Vertex v, String propname){\n@{ WikiPage wp=wikipage(v,propname)} image=\"@(wp.fulltext.replace(\"File:\",\"\"))\"}\n"
328         + "@def prop(Vertex v, String propname){\n"
329         + " @(v.property(propname).value().toString())\n" + "}\n"
330         + "<graphviz>\n" + "digraph hubandspoke @(\"{\")\n"
331         + "  edge [dir=\"both\"]\n" + "  layout=\"circo\";\n"
332         + "node [shape=circle,\n"
333         + "      fixedsize=true, # don't allow nodes to change sizes dynamically\n"
334         + "      width=1]\n" + "@for (Vertex mA:moduleVs) {\n"
335         + " @(prop(mA,\"name\")) [ @link(mA,\"wikipage\") @image(mA,\"logo\") ]\n"
336         + "}\n" + "@for (Vertex mA:moduleVs) {\n"
337         + "@for (Vertex mB:moduleVs) {\n" + " @if (mA!=mB) {\n"
338         + "@(prop(mA,\"name\")) ->@(prop(mB,\"name\"))\n" + "}\n" + "}\n"
339         + "}\n" + "@(\"}\")\n" + "</graphviz>\n";
340     String graphVizCode = rythmContext.render(template, rootMap);
341     // debug = true;
342     if (debug) {
343       System.out.println(graphVizCode);
344 
345       for (Vertex moduleV : moduleVs) {
346         System.out.println(String.format("# [[%s]]",
347             ((WikiPage) (moduleV.property("logo").value())).fulltext));
348       }
349     }
350   }
