55  /**
56   * Simple Network Monitoring Protocol
57   * see https://gist.github.com/akirad/5597203
58   * 
59   * @author wf
60   *
61   */
62  public class SNMP {
63    protected static Logger LOGGER = Logger
64        .getLogger("com.bitplan.simplegraph.snmp");
65    
66    private String targetAddr;
67    private String commStr;
68    private int snmpVersion;
69    private String portNum;
70    private int retries;
71    private int timeOutMSecs;
72  
73    static final private byte TAG1 = (byte) 0x9f;
74    static final private byte TAG_FLOAT = (byte) 0x78;
75    static final private byte TAG_DOUBLE = (byte) 0x79;
76    /**
77     * 
78     * @param targetAddr
79     * @throws IOException
80     */
81    public SNMP(String targetAddr, int retries, int timeOutMSecs)
82        throws IOException {
83      // Set default value.
84      this.targetAddr = targetAddr;
85      this.timeOutMSecs = timeOutMSecs;
86      this.retries = retries;
87      commStr = "public";
88      snmpVersion = SnmpConstants.version1;
89      portNum = "161";
90    }
91  
92    public List<VariableBinding> walk(String oidStr) throws IOException {
93      List<VariableBinding> bindings = new ArrayList<VariableBinding>();
94      Address targetAddress = GenericAddress
95          .parse("udp:" + targetAddr + "/" + portNum);
96      TransportMapping<? extends Address> transport = new DefaultUdpTransportMapping();
97      Snmp snmp = new Snmp(transport);
98      transport.listen();
99  
100     // setting up target
101     CommunityTarget target = new CommunityTarget();
102     target.setCommunity(new OctetString(commStr));
103     target.setAddress(targetAddress);
104     target.setRetries(retries);
105     target.setTimeout(timeOutMSecs);
106     target.setVersion(snmpVersion);
107     OID oid;
108     oid = new OID(oidStr);
109 
110     // Get MIB data.
111     TreeUtils treeUtils = new TreeUtils(snmp, new DefaultPDUFactory());
112     List<TreeEvent> events = treeUtils.getSubtree(target, oid);
113     if (events != null && events.size() > 0) {
114          // Handle the snmpwalk result.
115       for (TreeEvent event : events) {
116         if (event == null) {
117           continue;
118         }
119         if (event.isError()) {
120           System.err.println("oid [" + oid + "] " + event.getErrorMessage());
121           continue;
122         }
123 
124         VariableBinding[] varBindings = event.getVariableBindings();
125         if (varBindings == null || varBindings.length == 0) {
126           continue;
127         }
128         for (VariableBinding varBinding : varBindings) {
129           if (varBinding == null) {
130             continue;
131           }
132           bindings.add(varBinding);
133         }
134       }
135       snmp.close();
136     }
137     return bindings;
138   }
139   /**
140    * see https://github.com/fbacchella/jrds
141    * 
142    * @param variable
143    * @return
144    */
145   public Object getValue(Variable variable) {
146     Object retvalue = null;
147     if (variable != null) {
148       int type = variable.getSyntax();
149       if (variable instanceof OID) {
150         retvalue = variable;
151       } else if (variable instanceof UnsignedInteger32) {
152         if (variable instanceof TimeTicks) {
153           long epochcentisecond = variable.toLong();
154           retvalue = new Double(epochcentisecond / 100.0);
155         } else
156           retvalue = variable.toLong();
157       } else if (variable instanceof Integer32)
158         retvalue = variable.toInt();
159       else if (variable instanceof Counter64)
160         retvalue = variable.toLong();
161       else if (variable instanceof OctetString) {
162         if (variable instanceof Opaque) {
163           retvalue = resolvOpaque((Opaque) variable);
164         } else {
165           // It might be a C string, try to remove the last 0;
166           // But only if the new string is printable
167           OctetString octetVar = (OctetString) variable;
168           int length = octetVar.length();
169           if (length > 1 && octetVar.get(length - 1) == 0) {
170             OctetString newVar = octetVar.substring(0, length - 1);
171             if (newVar.isPrintable()) {
172               variable = newVar;
173               LOGGER.log(Level.INFO, "Convertion an octet stream from "
174                   + octetVar + " to " + variable);
175             }
176           }
177           retvalue = variable.toString();
178         }
179       } else if (variable instanceof Null) {
180         retvalue = null;
181       } else if (variable instanceof IpAddress) {
182         retvalue = ((IpAddress) variable).getInetAddress();
183       } else {
184         LOGGER.log(Level.WARNING,
185             "Unknown syntax " + AbstractVariable.getSyntaxString(type));
186       }
187     }
188     return retvalue;
189   }
190 
191   private Object resolvOpaque(Opaque var) {
192 
193     // If not resolved, we will return the data as an array of bytes
194     Object value = var.getValue();
195 
196     try {
197       byte[] bytesArray = var.getValue();
198       ByteBuffer bais = ByteBuffer.wrap(bytesArray);
199       BERInputStream beris = new BERInputStream(bais);
200       byte t1 = bais.get();
201       byte t2 = bais.get();
202       int l = BER.decodeLength(beris);
203       if (t1 == TAG1) {
204         if (t2 == TAG_FLOAT && l == 4)
205           value = new Float(bais.getFloat());
206         else if (t2 == TAG_DOUBLE && l == 8)
207           value = new Double(bais.getDouble());
208       }
209     } catch (IOException e) {
210       LOGGER.log(Level.SEVERE, var.toString());
211     }
212     return value;
213   }
