44  /**
45   * a slide show node wraps a power point slideshow
46   * 
47   * @author wf
48   *
49   */
50  public class SlideShowNode extends SimpleNodeImpl implements SlideShow {
51  
52    protected XMLSlideShow slideshow;
53    protected String pathOrUl;
54    protected File pptFile;
55  
56    /**
57     * get the SlideShow
58     */
59    public XMLSlideShow getSlideshow() {
60      return slideshow;
61    }
62  
63    /**
64     * create a SlideShow
65     * 
66     * @param simpleGraph
67     * @param nodeQuery
68     * @throws Exception
69     */
70    public SlideShowNode(SimpleGraph simpleGraph, String pathOrUrl,
71        String... keys) {
72      super(simpleGraph, "slideshow", keys);
73      InputStream is = null;
74      try {
75        try {
76          URL url = new URL(pathOrUrl);
77          is = url.openStream();
78        } catch (MalformedURLException e1) {
79          this.pathOrUl = pathOrUrl;
80          pptFile = new File(pathOrUl);
81          if (pptFile.canRead())
82            is = new FileInputStream(pathOrUl);
83        }
84        if (is != null)
85          slideshow = new XMLSlideShow(is);
86        else
87          slideshow = new XMLSlideShow();
88      } catch (IOException e) {
89        throw new RuntimeException(e);
90      }
91      super.setVertexFromMap();
92    }
93  
94    /**
95     * get the core properties
96     * 
97     * @return
98     */
99    public CoreProperties getCoreProperties() {
100     POIXMLProperties props = slideshow.getProperties();
101     CoreProperties cp = props.getCoreProperties();
102     return cp;
103   }
104 
105   @Override
106   public Map<String, Object> initMap() {
107     map.put("path", pathOrUl);
108     CoreProperties cp = getCoreProperties();
109     map.put("title", cp.getTitle());
110     return map;
111   }
112 
113   @Override
114   public Stream<SimpleNode> out(String edgeName) {
115     return inOrOut(edgeName);
116   }
117 
118   @Override
119   public Stream<SimpleNode> in(String edgeName) {
120     return inOrOut(edgeName);
121   }
122 
123   /**
124    * link me via the given edgeName
125    * @param edgeName
126    * @return - a stream of nodes
127    */
128   private Stream<SimpleNode> inOrOut(String edgeName) {
129     Stream<SimpleNode> links = Stream.of();
130     switch (edgeName) {
131     case "slides":
132       List<SimpleNode> slideNodes = new ArrayList<SimpleNode>();
133       List<XSLFSlide> slides = slideshow.getSlides();
134       int pageNo = 1;
135       for (XSLFSlide slide : slides) {
136         SlideNode slideNode = new SlideNode(this, slide);
137         slideNodes.add(slideNode);
138         slideNode.property("pageNo", pageNo++);
139         slideNode.property("pages", slides.size());
140         this.getVertex().addEdge("slides", slideNode.getVertex());
141         // redundant backpointer?
142         slideNode.getVertex().addEdge("slideshow", this.getVertex());
143       }
144       links = slideNodes.stream();
145       break;
146     }
147     return links;
148   }
149 
150   @Override
151   public String getTitle() {
152     Object titleObj = map.get("title");
153     if (titleObj != null)
154       return titleObj.toString();
155     else
156       return "";
157   }
158 
159   @Override
160   public void setTitle(String title) {
161     this.getCoreProperties().setTitle(title);
162     map.put("title", title);
163   }
164 
165   @Override
166   public void save() throws Exception {
167     FileOutputStream out = new FileOutputStream(pathOrUl);
168     slideshow.write(out);
169     out.close();
170     slideshow.close();
171   }
172 
173   @Override
174   public Slide createSlide() {
175     XSLFSlide xslide = slideshow.createSlide();
176     return new SlideNode(this, xslide);
177   }
178 
179 }
