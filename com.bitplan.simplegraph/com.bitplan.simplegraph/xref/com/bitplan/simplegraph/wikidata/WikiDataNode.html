99    @SuppressWarnings("unchecked")
100   @Override
101   public Map<String, Object> initMap() {
102     DataObjectFactoryImpl df = new DataObjectFactoryImpl();
103     if (doc instanceof TermedDocument) {
104       TermedDocument tdoc = ((TermedDocument) doc);
105       for (String languageCode : getSystem().getLanguages()) {
106         MonolingualTextValue langlabel = tdoc.getLabels().get(languageCode);
107         if (langlabel != null) {
108           String langValue = df
109               .getStringValue(tdoc.getLabels().get(languageCode).getText())
110               .getString();
111           map.put("label_" + languageCode, langValue);
112         }
113       }
114     }
115     if (doc instanceof ItemDocument) {
116       ItemDocument idoc = ((ItemDocument) doc);
117       Map<String, SiteLink> sitelinks = idoc.getSiteLinks();
118       for (String languageCode : getSystem().getLanguages()) {
119         SiteLink languageWikiLink = sitelinks.get(languageCode + "wiki");
120         if (languageWikiLink != null) {
121           map.put("wiki_" + languageCode,
122               df.getStringValue(languageWikiLink.getPageTitle()));
123         }
124       }
125     }
126 
127     if (doc instanceof StatementDocument) {
128       StatementDocument sdoc = ((StatementDocument) doc);
129 
130       // loop over all statements
131       Iterator<Statement> siterator = sdoc.getAllStatements();
132       EntityIdValue subject = null;
133       // move over all statements
134       while (siterator.hasNext()) {
135         // get the current statement
136         Statement s = siterator.next();
137         // get the claim of the Statement
138         Claim claim = s.getClaim();
139         if (subject == null) {
140           subject = claim.getSubject(); // Subject - TODO - check under which
141                                         // circumstances there might be
142                                         // different ones
143           map.put("wikidata_id", subject.getId());
144           map.put("wikidata_type", subject.getEntityType());
145         }
146         Snak snak = claim.getMainSnak();
147         PropertyIdValue propId = snak.getPropertyId();
148         this.entityIdById.put(propId.getId(), propId);
149         Value value = snak.getValue();
150 
151         // handle properties with multiple values
152         // like child
153         if (map.containsKey(propId.getId())) {
154           // get the current property
155           Object prop = map.get(propId.getId());
156           // prepare for multiple entries
157           List<Object> propList = null;
158           // is the propery a list already?
159           if (prop instanceof List) {
160             propList = (List<Object>) prop;
161           } else {
162             // replace single element with list
163             propList = new ArrayList<Object>();
164             propList.add(prop);
165             map.put(propId.getId(), propList);
166           }
167           propList.add(value);
168           // System.out.println(propId);
169         } else {
170           // first time single value
171           map.put(propId.getId(), value);
172         }
173         // optionally cache values
174         // first the property
175         SimpleNode propNode = this.getSystem().cache(propId, true);
176         if (propNode != null) {
177           String propLabel = propNode.getMap().get("label_en").toString();
178           if (propLabel != null) {
179             this.entityIdByName.put(propLabel, propId);
180           }
181         }
182         // potentially the value
183         // https://www.mediawiki.org/wiki/Wikibase/DataModel#Datatypes_and_their_Values
184         // https://wikidata.github.io/Wikidata-Toolkit/org/wikidata/wdtk/datamodel/interfaces/EntityIdValue.html
185         if (value instanceof ItemIdValue) {
186           ItemIdValue itemIdValue = (ItemIdValue) value;
187           this.entityIdById.put(itemIdValue.getId(), itemIdValue);
188           boolean cacheOptional = true;
189           if (keys.getKeysList().isPresent())
190             cacheOptional = !keys.hasKey(propId.getId());
191           SimpleNode itemNode = this.getSystem().cache(itemIdValue,
192               cacheOptional);
193         }
194       }
195     }
196     return map;
197   }

199   /**
200    * get the property value for the given key
201    * 
202    * @param key
203    * @return - the property value
204    */
205   public Object getProperty(String key) {
206     String propId = key;
207     if (!key.startsWith("P") && entityIdByName.containsKey(key)) {
208       propId = entityIdByName.get(key).getId();
209     }
210     Object value = map.get(propId);
211     return typeConvert(value);
212   }

214   /**
215    * convert the types according to
216    * https://www.mediawiki.org/wiki/Wikibase/DataModel#Datatypes_and_their_Values
217    * 
218    * @param value
219    * @return the converted type
220    */
221   public Object typeConvert(Object value) {
222     if (value instanceof StringValue) {
223       StringValue svalue = (StringValue) value;
224       return svalue.getString();
225     } else if (value instanceof TimeValue) {
226       TimeValue timeValue=(TimeValue) value;
227       int year=(int) timeValue.getYear();
228       int month=timeValue.getMonth();
229       int day=timeValue.getDay();
230       int hour=timeValue.getHour();
231       int minute=timeValue.getMinute();
232       int sec=timeValue.getSecond();
233       GregorianCalendar timeDate=new GregorianCalendar(year,month,day,hour,minute,sec);
234       // TODO use timeValue.getPrecision() to select display
235       return this.isoDateFormat.format(timeDate.getTime());
236     } else if (value instanceof ItemIdValue) {
237       ItemIdValue itemIdValue = (ItemIdValue) value;
238       String itemId = itemIdValue.getId();
239       String valueStr = itemId;
240       SimpleNode itemNode = getSystem().cache(itemIdValue, true);
241       if (itemNode != null) {
242         Map<String, Object> itemMap = itemNode.getMap();
243         if (itemMap.containsKey("label_en")) {
244           valueStr = itemMap.get("label_en").toString()+" ("+itemId+")";
245         }
246       }
247       return valueStr;
248     }
249     return value;
250   }

258   /**

264   public SimpleNode getEntityNode(JacksonValueItemId itemValue) {
265     SimpleNode node = this.getSystem().moveTo(itemValue.getId(),keys.getKeys());

267   }

269   @Override
270   public Stream<SimpleNode> out(String edgeName) {
271     List<SimpleNode> outs = new ArrayList<SimpleNode>();
272     String propertyId = edgeName;
273     if (!edgeName.matches("P[0-9]+")) {
274       EntityIdValue entityIdValue = this.entityIdByName.get(edgeName);
275       if (entityIdValue != null) {
276         propertyId = entityIdValue.getId();
277       }
278     }
279     if (map.containsKey(propertyId)) {
280       Object outValue = map.get(propertyId);
281       if (outValue instanceof JacksonValueItemId) {
282         outs.add(getEntityNode((JacksonValueItemId) outValue));
283       } else {
284         @SuppressWarnings("unchecked")
285         List<JacksonValueItemId> entityValues = (List<JacksonValueItemId>) outValue;
286         entityValues
287             .forEach(entityValue -> outs.add(getEntityNode(entityValue)));
288       }
289     }
290     return outs.stream();
291   }

301   // show name values
302   public void printNameValues(PrintStream out) {
303     Map<String, Object> map = this.getMap();
304     for (String key : map.keySet()) {
305       if (this.keys.hasKey(key)) {
306         String label = key;
307         EntityIdValue entityId = this.entityIdById.get(key);
308         if (entityId != null) {
309           SimpleNode entityNode = getSystem().cache(entityId, true);
310           if (entityNode != null) {
311             label = entityNode.getMap().get("label_en").toString();
312           }
313         }
314         String valueStr=this.getProperty(key).toString();
315         out.println(
316             String.format("%s (%s) = %s", label, key, valueStr));
317       }
318     }
