21  package com.bitplan.simplegraph.powerpoint;
22  
23  import java.awt.Color;
24  import java.awt.Dimension;
25  import java.awt.Graphics2D;
26  import java.awt.RenderingHints;
27  import java.awt.geom.Rectangle2D;
28  import java.awt.image.BufferedImage;
29  import java.io.File;
30  import java.io.FileOutputStream;
31  import java.io.OutputStream;
32  
33  import org.apache.poi.xslf.usermodel.XMLSlideShow;
34  import org.apache.poi.xslf.usermodel.XSLFSlide;
35  
36  /**
37   * wrapper for a Slide's image
38   * @author wf
39   *
40   */
41  public class SlideImage {
42    transient BufferedImage image;
43    transient public Graphics2D graphics;
44    double zoom;
45    transient XSLFSlide slide;
46    int width;
47    int height;
48    File imageFile;
49    transient static XMLSlideShow ppt = null;
50    
51    /**
52     * create the slideImage
53     * @param slide
54     * @param width
55     * @param height
56     * @param zoom
57     */
58    public SlideImage(XSLFSlide slide,int width,int height,double zoom) {
59      this.slide=slide;
60      this.width=width;
61      this.height=height;
62      this.zoom=zoom;
63    }
64    
65    /**
66     * prepare image for the given width and height with the given zoom factor
67     * 
68     * @param width
69     *          - in pixel
70     * @param height
71     *          - in pixel
72     * @param zoom
73     *          - zoom factor for e.g. magnification
74     */
75    public void prepareImage() {
76      Dimension dim = new Dimension(width,height);
77      image = new BufferedImage((int) Math.ceil(dim.width * zoom),
78          (int) Math.ceil(dim.height * zoom), BufferedImage.TYPE_INT_RGB);
79      graphics = image.createGraphics();
80      graphics.scale(zoom, zoom);
81      graphics.setPaint(Color.LIGHT_GRAY);
82      // see http://grepcode.com/file/repo1.maven.org/maven2/org.apache.poi/poi-ooxml/3.10.1/org/apache/poi/xslf/util/PPTX2PNG.java
83      graphics.setRenderingHint(RenderingHints.KEY_ANTIALIASING, RenderingHints.VALUE_ANTIALIAS_ON);
84      graphics.setRenderingHint(RenderingHints.KEY_RENDERING, RenderingHints.VALUE_RENDER_QUALITY);
85      graphics.setRenderingHint(RenderingHints.KEY_INTERPOLATION, RenderingHints.VALUE_INTERPOLATION_BICUBIC);
86      graphics.setRenderingHint(RenderingHints.KEY_FRACTIONALMETRICS, RenderingHints.VALUE_FRACTIONALMETRICS_ON);
87      graphics.setColor(Color.white);
88      graphics.clearRect(0, 0, width, height);
89      graphics.fill(new Rectangle2D.Float(0, 0, dim.width, dim.height));
90    }
91    
92    private void draw(XSLFSlide slide, Graphics2D graphics) {
93      if (ppt==null)
94        ppt=new XMLSlideShow();
95      XSLFSlide drawSlide = ppt.createSlide();
96      drawSlide.importContent(slide);
97      drawSlide.draw(graphics);
98    }
99    
100   /**
101    * draw the image 
102    * @param withBackground
103    * @return 
104    */
105   public BufferedImage drawImage(boolean withBackground) {
106     prepareImage();
107     if (withBackground)
108       slide.draw(graphics);
109     else
110       draw(slide, graphics);
111     return image;
112   }
113   
114   /**
115    * save me to the given file path
116    * @param filepath
117    * @return the file
118    * @throws Exception
119    */
120   public File save(String filepath) throws Exception {
121     imageFile = new File(filepath);
122     FileOutputStream out = new FileOutputStream(imageFile);
123     save(out);
124     return imageFile;
125   }
126   
127   /**
128    * save the image to the given outputstream
129    * @param out
130    * @throws Exception
131    */
132   public void save(OutputStream out) throws Exception {
133     javax.imageio.ImageIO.write(image, "png", out);
134     out.close();
135   }
136   
137 }
