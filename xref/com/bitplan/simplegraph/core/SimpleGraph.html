21  package com.bitplan.simplegraph.core;
22  
23  import java.util.function.Consumer;
24  
25  import org.apache.tinkerpop.gremlin.process.traversal.dsl.graph.GraphTraversalSource;
26  import org.apache.tinkerpop.gremlin.structure.Graph;
27  import org.apache.tinkerpop.gremlin.structure.Vertex;
28  
29  /**
30   * The Simple Graph interface is inspired by the mathematical concept of a
31   * graph: https://en.wikipedia.org/wiki/Graph_(discrete_mathematics)
32   * 
33   * it wraps an Apache Tinkerpop / Gremlin Graph
34   * and allows generic graph traversals
35   * 
36   * a graph can also be used in cases where you'd normally use lists, tables and trees - the graph
37   * is the more general concept
38   *
39   * @author wf
40   *
41   */
42  public interface SimpleGraph {
43  
44    // access my wrapped graph
45    public Graph graph();
46      
47    // these are the interfaces with default implementations
48    // convenience method to access to Gremlin wrapped objects,
49    public default GraphTraversalSource g() {
50      return this.graph().traversal();
51    }
52    
53    public Vertex addVertex(SimpleNode other); 
54  
55    // SimpleGraph API
56    /**
57     * get the "start" node of this graph
58     * @return the start node
59     */
60    public SimpleNode getStartNode();
61   
62    /**
63     * set the "start" node of this graph
64     * @param startNode
65     * @return
66     */
67    public SimpleNode setStartNode(SimpleNode startNode);
68    
69    public boolean isDebug();
70  
71    public void setDebug(boolean debug);
72    
73    /**
74     * forAll
75     */
76    public default void forAll(Consumer<Vertex> consumer) {
77      g().V().forEachRemaining(consumer);
78    };
79  
80  }
