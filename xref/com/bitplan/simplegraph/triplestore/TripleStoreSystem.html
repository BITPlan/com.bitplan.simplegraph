21  package com.bitplan.simplegraph.triplestore;
22  
23  import java.io.File;
24  
25  import org.sidif.triple.Triple;
26  import org.sidif.triple.TripleQuery;
27  import org.sidif.triple.TripleStore;
28  import org.sidif.util.TripleStoreBuilder;
29  
30  import com.bitplan.simplegraph.core.SimpleNode;
31  import com.bitplan.simplegraph.core.SimpleSystem;
32  import com.bitplan.simplegraph.impl.SimpleSystemImpl;
33  
34  public class TripleStoreSystem extends SimpleSystemImpl {
35    TripleStore tripleStore;
36    TripleQuery query;
37  
38    @Override
39    public SimpleNode moveTo(String nodeQuery, String... keys) {
40      String[] parts = nodeQuery.split("=");
41      String predicate = parts[0];
42      String object = parts[1];
43      Triple triple = query.selectSingle(null, predicate, object);
44      SimpleNode node = new TripleNode(this, triple, keys);
45      return node;
46    }
47  
48    @Override
49    public SimpleSystem connect(String... params) throws Exception {
50      if (params.length != 1)
51        throw new IllegalArgumentException(
52            "only one connection parameter (path to SiDIF file) allowed at this time");
53      File sidifFile = new File(params[0]);
54      tripleStore = TripleStoreBuilder.fromFile(sidifFile);
55      query = tripleStore.query();
56      return this;
57    }
58  
59    public TripleStoreSystem() {
60      super.setName("SiDIF TripleStore");
61      super.setVersion("0.0.8");
62    }
63  
64    @Override
65    public Class<? extends SimpleNode> getNodeClass() {
66      return TripleNode.class;
67    }
68  
69  }
