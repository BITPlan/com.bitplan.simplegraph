21  package com.bitplan.simplegraph.impl;
22  
23  import org.apache.tinkerpop.gremlin.structure.Graph;
24  import org.apache.tinkerpop.gremlin.structure.Vertex;
25  import org.apache.tinkerpop.gremlin.tinkergraph.structure.TinkerGraph;
26  
27  import com.bitplan.simplegraph.core.SimpleGraph;
28  import com.bitplan.simplegraph.core.SimpleNode;
29  
30  /**
31   * default implementation of SimpleGraph interface - wraps a TinkerPop 3 graph
32   * @author wf
33   *
34   */
35  public class SimpleGraphImpl implements SimpleGraph {
36    protected transient boolean debug = false;
37  
38    public boolean isDebug() {
39      return debug;
40    }
41  
42    public void setDebug(boolean debug) {
43      this.debug = debug;
44    }
45    // The TinkerPop 3 graph
46    Graph graph;
47    
48    SimpleNode startNode;
49    
50    public SimpleGraphImpl() {
51      this(null); 
52    }
53    
54    /**
55     * construct me from the given simple Graph
56     * @param simpleGraph
57     */
58    public SimpleGraphImpl(SimpleGraph simpleGraph) {
59      if (simpleGraph==null)
60        graph = TinkerGraph.open(); 
61      else
62        graph=simpleGraph.graph();
63    }
64    
65    @Override
66    public SimpleNode getStartNode() {
67      return startNode;
68    }
69  
70    @Override
71    public SimpleNode setStartNode(SimpleNode startNode) {
72      this.startNode = startNode;
73      return startNode;
74    }
75    
76    @Override
77    public Graph graph() {
78      return graph;
79    }
80    
81    @Override
82    public Vertex addVertex(SimpleNode other) {
83      if (other instanceof SimpleNodeImpl) {
84        SimpleNodeImpl sother = (SimpleNodeImpl) other;
85        Vertex vertex=graph.addVertex(sother.keyValueList.toArray());
86        return vertex;
87      }
88      throw new IllegalArgumentException("addVertex needs a SimpleNodeImpl argument but got "+other.getClass().getName());
89    }
90  
91  }
