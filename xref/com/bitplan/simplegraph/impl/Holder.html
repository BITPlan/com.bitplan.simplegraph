21  package com.bitplan.simplegraph.impl;
22  
23  import java.util.ArrayList;
24  import java.util.List;
25  
26  /**
27   * generic holder to be use in e.g. in lambdas to access outer scope variables which need to be final
28   * @author wf
29   *
30   * @param <T>
31   */
32  public class Holder<T> {
33    private List<T> values=new ArrayList<T>();
34    private Class<T> clazz;
35    
36    public Holder() {
37    }
38    
39    public boolean isPresent() {
40      return values.size()>0;
41    }
42    
43    /**
44     * construct this holder from a value
45     * @param value
46     */
47    public Holder(T value) {
48      values.add(value);
49      clazz=(Class<T>) value.getClass();
50    }
51    
52    /**
53     * get the first held value or null if it is not present
54     * @return the first value
55     */
56    public T getFirstValue() {
57      if (isPresent())
58        return values.get(0);
59      else
60        return null;
61    }
62    
63    public List<T> getValues() {
64      return values;
65    }
66    
67    public void setValue(T value) {
68      values.clear();
69      add(value);
70    }
71  
72    public void add(T value) {
73      values.add(value);
74    }
75  
76    /**
77     * add all holded values from the other holder
78     * @param other
79     */
80    public void addAll(Holder<T> other) {
81      other.getValues().forEach(otherT->this.add(otherT));
82    }
83    public static <T> T convertInstanceOfObject(Object o, Class<T> clazz) {
84      try {
85        return clazz.cast(o);
86    } catch(ClassCastException e) {
87        return null;
88    }
89    }
90  
91    public void increment() {
92      if (this.isPresent()) {
93        Object o=this.getFirstValue();
94        if (o instanceof Integer) {
95          int i=(Integer)o;
96          Integer i1 = i+1;
97          this.setValue(convertInstanceOfObject(i1,clazz));
98        }
99      }
100   }
101 }
