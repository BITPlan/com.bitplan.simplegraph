74    /**
75     * get the contents of a sheet
76     * 
77     * @param sheet
78     * @return
79     */
80    public List<List<Object>> getSheetContent(XSSFSheet sheet) {
81      List<List<Object>> result = new ArrayList<List<Object>>();
82      Iterator<Row> rows = sheet.rowIterator();
83      while (rows.hasNext()) {
84        XSSFRow row = (XSSFRow) rows.next();
85        Iterator<Cell> cells = row.cellIterator();
86        List<Object> rowList = new ArrayList<Object>();
87        while (cells.hasNext()) {
88          XSSFCell cell = (XSSFCell) cells.next();
89          Object cellValue = getCellValue(cell);
90          rowList.add(cellValue);
91        }
92        if (rowList.size() > 0)
93          result.add(rowList);
94      }
95      return result;
96    }

98    private Object getCellValue(XSSFCell cell) {
99      Object cellValue = null;
100     CellType cellType = cell.getCellTypeEnum();
101     if (CellType.FORMULA == cellType)
102       cellType = cell.getCachedFormulaResultTypeEnum();
103     switch (cellType) {
104     case BOOLEAN:
105       cellValue = cell.getBooleanCellValue();
106       break;
107     case NUMERIC:
108       cellValue = cell.getNumericCellValue();
109       break;
110     case STRING:
111       cellValue = cell.getStringCellValue();
112       break;
113     case BLANK:
114       break;
115     case ERROR:
116       cellValue = cell.getErrorCellValue();
117       break;
118 
119     // CELL_TYPE_FORMULA will never occur
120     case FORMULA:
121       break;
122     case _NONE:
123       cellValue = cell.toString();
124       break;
125     default:
126       break;
127     }
128     return cellValue;
129   }

147   /**
148    * set the cellComment for the given cell to the given text see
149    * https://stackoverflow.com/q/16099912/1497139
150    * 
151    * @param cell
152    * @param text
153    */
154   @SuppressWarnings("rawtypes")
155   public static void setComment(Cell cell, String text) {
156     Drawing drawing = cell.getSheet().createDrawingPatriarch();
157     CreationHelper factory = cell.getSheet().getWorkbook().getCreationHelper();
158     ClientAnchor anchor = factory.createClientAnchor();
159     anchor.setCol1(cell.getColumnIndex());
160     anchor.setCol2(cell.getColumnIndex() + 1);
161     anchor.setRow1(cell.getRowIndex());
162     anchor.setRow2(cell.getRowIndex() + 3);
163 
164     Comment comment = drawing.createCellComment(anchor);
165     RichTextString str = factory.createRichTextString(text);
166     comment.setVisible(Boolean.TRUE);
167     comment.setString(str);
168 
169     cell.setCellComment(comment);
170   }

172   /**
173    * create a Workbook for the given graph
174    * 
175    * @param g
176    * @return
177    */
178   public static Workbook createWorkBook(GraphTraversalSource g) {
179     // https://poi.apache.org/spreadsheet/quick-guide.html#NewSheet
180     Workbook wb = new XSSFWorkbook();
181     // add a sheet of vertices per vertex Label
182     g.V().label().dedup()
183         .forEachRemaining(vertexLabel -> addSheet(wb, vertexLabel, g.V()));
184     // add a sheet of edges per edge Label
185     g.E().label().dedup()
186         .forEachRemaining(edgeLabel -> addSheet(wb, edgeLabel, g.E()));
187     return wb;
188   }

190   /**
191    * add a sheet for the given label and GraphTraversal (Vertex/Edge)
192    * 
193    * @param wb
194    * @param itemLabel
195    * @param items
196    */
197   private static <T> void addSheet(Workbook wb, String itemLabel,
198       GraphTraversal<T, T> items) {
199     boldStyle = wb.createCellStyle();
200     Font font = wb.createFont();
201     font.setBold(true);
202     boldStyle.setFont(font);
203 
204     // create one sheet per Label
205     Sheet sheet = wb.createSheet(fixSheetName(itemLabel));
206     // rowIndex to be used in Lambda starting from 0
207     Holder<Integer> rowIndex = new Holder<Integer>(0);
208     // look for nodes/vertices with the given vertexLabel
209     items.hasLabel(itemLabel).forEachRemaining(item -> {
210       // get the current rowNumber
211       Integer rowNumber = rowIndex.getFirstValue();
212       // create a new row for this rowNumber
213       Holder<Row> rowHolder = new Holder<Row>(sheet.createRow(rowNumber));
214       // first row is header line
215       if (rowNumber == 0) {
216         // define style
217         Row headerRow = rowHolder.getFirstValue();
218         // add cells for the given row
219         Holder<Integer> colIndex = new Holder<Integer>(0);
220         if (item instanceof Vertex) {
221           addCell(wb, headerRow, "id", colIndex, boldStyle);
222           Vertex vertex = (Vertex) item;
223           addCells(wb, headerRow, vertex, colIndex, true);
224         } else if (item instanceof Edge) {
225           Edge edge = (Edge) item;
226           edge.properties().forEachRemaining(prop -> {
227             addCell(wb, headerRow, edge.label(), colIndex, boldStyle);
228           });
229           addCell(wb, headerRow, "in", colIndex, boldStyle);
230           addCell(wb, headerRow, "out", colIndex, boldStyle);
231         }
232         // create a new Row
233         rowHolder.setValue(sheet.createRow(++rowNumber));
234         rowIndex.setValue(rowNumber);
235       }
236       Row row = rowHolder.getFirstValue();
237       Holder<Integer> colIndex = new Holder<Integer>(0);
238       if (item instanceof Vertex) {
239         Vertex vertex = (Vertex) item;
240         addCell(wb, row, vertex.id(), colIndex, null);
241         addCells(wb, row, vertex, colIndex, false);
242       } else if (item instanceof Edge) {
243         Edge edge = (Edge) item;
244         edge.properties().forEachRemaining(prop -> {
245           addCell(wb, row, prop.value(), colIndex, null);
246         });
247         addCell(wb, row, edge.inVertex().id(), colIndex, null);
248         addCell(wb, row, edge.outVertex().id(), colIndex, null);
249       }
250       rowIndex.setValue(rowIndex.getFirstValue() + 1);
251     });
252   }

254   /**
255    * add the cells for the given vertex to the given row
256    * 
257    * @param wb
258    * 
259    * @param row
260    * @param vertex
261    * @param colIndex
262    * @param header
263    */
264   private static void addCells(Workbook wb, Row row, Vertex vertex,
265       Holder<Integer> colIndex, boolean header) {
266     SimpleNode simpleNode = SimpleNode.of(vertex);
267     boolean done = false;
268     // do we have SimpleNode with a predefined order of keys?
269     if (simpleNode != null) {
270       Keys keys = simpleNode.getKeys();
271       if (!keys.isEmpty()) {
272         for (String key : keys.getKeysList().get()) {
273           Object value = null;
274           if (vertex.property(key).isPresent())
275             value = vertex.property(key).value();
276           addCell(wb, row, header ? key : value, colIndex,
277               header ? boldStyle : null);
278         }
279         // ordered mode done
280         done = true;
281       }
282     }
283     // their is no order - try our luck with the set of properties
284     if (!done) {
285       vertex.properties().forEachRemaining(prop -> {
286         if (prop.label() != SimpleNode.SELF_LABEL)
287           addCell(wb, row, header ? prop.label() : prop.value(), colIndex,
288               header ? boldStyle : null);
289       });
290     }
291   }

293   /**
294    * https://stackoverflow.com/questions/451452/valid-characters-for-excel-sheet-names
295    * 
296    * @param sheetName
297    * @return a valid sheetName
298    */
299   private static String fixSheetName(String sheetName) {
300     String invalid = "[]*/\\?";
301     for (int i = 0; i < invalid.length(); i++) {
302       char invalidChar = invalid.charAt(i);
303       sheetName = sheetName.replace(invalidChar, '_');
304     }
305     return sheetName;
306   }

321   /**
322    * add a Cell with the given value at the given column Index to the given row
323    * if a cellStyle is given the the cell's style to it
324    * 
325    * @param wb
326    * 
327    * @param row
328    * @param value
329    * @param colIndex
330    * @param cellStyle
331    */
332   private static void addCell(Workbook wb, Row row, Object value,
333       Holder<Integer> colIndex, CellStyle cellStyle) {
334     Cell cell = row.createCell(colIndex.getFirstValue());
335     CreationHelper createHelper = wb.getCreationHelper();
336     ;
337     colIndex.setValue(colIndex.getFirstValue() + 1);
338     if (value instanceof Integer) {
339       Integer intValue = (Integer) value;
340       cell.setCellValue(intValue.doubleValue());
341     } else if (value instanceof Long) {
342       Long longValue = (Long) value;
343       cell.setCellValue(longValue.doubleValue());
344     } else if (value instanceof Double) {
345       cell.setCellValue((Double) value);
346     } else if (value instanceof Date) {
347       if (cellStyle == null)
348         cellStyle = wb.createCellStyle();
349       cellStyle.setDataFormat(
350           createHelper.createDataFormat().getFormat(DATE_FORMAT));
351       Date date = (Date) value;
352       cell.setCellValue(date);
353     } else if (value instanceof Boolean) {
354       Boolean bool = (Boolean) value;
355       cell.setCellValue(bool);
356     } else {
357       if (value != null)
358         cell.setCellValue(value.toString()); // type handling!
359     }
360     if (cellStyle != null)
361       cell.setCellStyle(cellStyle);
362     if (debug) {
363       // add comments with type information
364       setComment(cell, value.getClass().getName());
365     }
366   }
