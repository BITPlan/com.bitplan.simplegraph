51    @Test
52    public void testFileSystemTree() throws Exception {
53      SimpleSystem fs=new FileSystem();
54      assertEquals("FileSystem",fs.getName());
55      assertEquals("0.0.1",fs.getVersion());
56      FileNode start = (FileNode) fs.connect("").moveTo("../simplegraph-filesystem/src");
57      if (debug)
58        start.printNameValues(System.out);
59      start.recursiveOut("files",Integer.MAX_VALUE).forEach(childFile->{
60        if (debug)
61          childFile.printNameValues(System.out);
62      });
63      long filecount = start.g().V().count().next().longValue();
64      if (debug)
65        LOGGER.log(Level.INFO,""+filecount);
66      assertEquals(17,filecount);
67      GraphTraversal<Vertex, Vertex> javaFiles = start.g().V().has("ext", "java");
68      long javaFileCount=javaFiles.count().next().longValue();
69      assertEquals(4,javaFileCount);
70      javaFiles = start.g().V().has("ext", "java");
71      if (debug)
72        javaFiles.forEachRemaining(SimpleNode.printDebug);
73    }
74    /**
75     * get a File node for the given path
76     * 
77     * @param path
78     * @return
79     * @throws Exception
80     */
81    public static SimpleNode getFileNode(String path, int levels) throws Exception {
82      // create a new FileSystem access supplying the result as a SimpleSystem API
83      // the "global" variable "fs" is set as a side effect
84      fs = new FileSystem();
85      // connect to this system with no extra information (e.g. no credentials)
86      // and move to the path node
87      SimpleNode start = fs.connect("").moveTo(path);
88      // do gremlin style out traversals recursively to the given depth
89      start.recursiveOut("files", levels);
90      return start;
91    }
92    
93    @Test
94    public void testFullyQualifiedPath() throws Exception {
95      // debug=true;
96      SimpleNode start=getFileNode("../simplegraph-filesystem/src",2);
97      long nodeCount=start.g().V().count().next().longValue();
98      assertEquals(7,nodeCount);
99      if (debug) {
100       LOGGER.log(Level.INFO,"src has "+nodeCount+" subdirectories on the next two levels");
101       start.forAll(SimpleNode.printDebug);
102     }
103   }

105   @Test
106   public void testFileSystemParent() throws Exception {
107     // debug=true;
108     // starting from the file "src/etc/header" move up the hierarchy to steps
109     SimpleSystem fs=new FileSystem();
110     FileNode start = (FileNode) fs.connect("").moveTo("src/etc/header.txt");
111     FileNode src=(FileNode) start.in("parent").findFirst().get().in("parent").findFirst().get();
112     if (debug)
113       src.printNameValues(System.out);
114     assertEquals("src",src.getMap().get("name"));
115     assertEquals("src",src.getVertex().property("name").value());
116   }

125   @Test
126   public void testInvalidEdgeName() throws Exception  {
127     SimpleSystem fs=new FileSystem();
128     FileNode start = (FileNode) fs.connect("").moveTo("src/test");
129     // link is an invalid edge for Files
130     try {
131       start.out("link");
132       fail("Exception expected");
133     } catch (Throwable th) {
134       // debug=true;
135       if (debug)
136         System.out.println(th.getMessage());
137       assertEquals("unknown edgeName link",th.getMessage());
138     }
139   }

141   @Test
142   public void testRegularExpession() throws Exception {
143     SimpleNode start=getFileNode("src/test/java",3);
144     start.g().V().has("ext","java").has("name",RegexPredicate.regex("Test.*")).forEachRemaining(node->{
145       FileNode fileNode=(FileNode) node.property("mysimplenode").value(); 
146       if (debug)
147         fileNode.printNameValues(System.out);
148       assertTrue(fileNode.getMap().get("name").toString().startsWith("Test"));
149     });
150   }
