21  package com.bitplan.simplegraph.filesystem;
22  
23  import java.io.File;
24  import java.text.SimpleDateFormat;
25  import java.util.ArrayList;
26  import java.util.List;
27  import java.util.Map;
28  import java.util.stream.Stream;
29  
30  import org.apache.commons.io.FilenameUtils;
31  
32  import com.bitplan.simplegraph.core.SimpleNode;
33  import com.bitplan.simplegraph.impl.SimpleNodeImpl;
34  
35  /**
36   * a File in the File system
37   * 
38   * @author wf
39   *
40   */
41  public class FileNode extends SimpleNodeImpl {
42  
43    // File to be wrapped in a Node
44    File file;
45    // System to be used
46    FileSystem fileSystem;
47    
48    transient String ext;
49    
50    final public static SimpleDateFormat isoDateFormat = new SimpleDateFormat(
51        "yyyy-MM-dd HH:mm:ss");
52  
53    /**
54     * default constructor for cache handling
55     * @param fileSystem
56     * @param keys
57     */
58    public FileNode(FileSystem fileSystem,String ... keys) {
59      super(fileSystem,"file",keys);
60    }
61    
62    /**
63     * create me from a path
64     * @param fileSystem - the FileSystem to create the FileNode for
65     * @param path
66     * @param keys 
67     */
68    public FileNode(FileSystem fileSystem,String path, String ... keys) {
69      this(fileSystem,new File(path),keys);
70    }
71  
72    /**
73     * create me from a file
74     * @param fileSystem
75     * @param file
76     */
77    public FileNode(FileSystem fileSystem,File file,String ...keys) {
78      super(fileSystem,"file",keys);
79      this.fileSystem=fileSystem;
80      if (file!=null) {
81        this.file = file;
82        this.ext = FilenameUtils.getExtension(file.getName());
83      }
84      super.setVertexFromMap();
85    }
86  
87    @Override
88    public Map<String, Object> initMap() {
89      map.put("file", file);
90      map.put("path", file.getPath());
91      map.put("name", file.getName());
92      map.put("size", file.length());
93      map.put("ext", ext);
94      map.put("lastModified", isoDateFormat.format(file.lastModified()));
95      return map;
96    }
97  
98    @Override
99    public Stream<SimpleNode> out(String edgeName) {
100     return inOrOut(edgeName);
101   }
102   
103   @Override
104   public Stream<SimpleNode> in(String edgeName) {
105     return inOrOut(edgeName);
106   }
107 
108   protected Stream<SimpleNode> inOrOut(String edgeName) {
109     Stream<SimpleNode> links = Stream.of();
110     switch (edgeName) {
111     case "parent":
112       FileNode parent = new FileNode(fileSystem,file.getParentFile());
113       knit(parent,this);
114       links = Stream.of(parent);
115       break;
116     case "files":
117       if (file.isDirectory()) {
118         List<SimpleNode> files = new ArrayList<SimpleNode>();
119         for (File childFile : file.listFiles()) {
120           FileNode childFileNode=new FileNode(fileSystem,childFile);
121           knit(this,childFileNode);
122           files.add(childFileNode);
123         }
124         links = files.stream();
125       }
126       break;
127       default:
128         throw new RuntimeException("unknown edgeName "+edgeName);
129     }
130     return links;
131   }
132   
133   protected void knit(FileNode parent,FileNode child) {
134     child.getVertex().addEdge("parent", parent.getVertex());
135     parent.getVertex().addEdge("files", child.getVertex());
136   }
137 
138 }
