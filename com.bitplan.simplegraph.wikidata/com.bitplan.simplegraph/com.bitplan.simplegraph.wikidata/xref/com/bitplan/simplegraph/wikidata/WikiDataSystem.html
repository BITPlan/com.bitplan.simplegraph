52    public static final String PURPOSE_PROPERTY = "http://www.wikidata.org/ontology#Property";
53    public static final String PURPOSE_ITEM = "http://www.wikidata.org/ontology#Item";
54    transient WikibaseDataFetcher wbdf;
55    List<String> languages = new ArrayList<String>();
56  
57    public static String siteiri = "http://www.wikidata.org/entity/";
58    transient static boolean debug = true;
59    transient protected static Logger LOGGER = Logger
60        .getLogger("com.bitplan.wikidata");

62    @Override
63    public SimpleNode moveTo(String entityId, String... keys) {
64      if (wbdf == null)
65        throw new IllegalStateException("not connected");
66      EntityDocument entityDocument;
67      try {
68        entityDocument = wbdf.getEntityDocument(entityId);
69      } catch (MediaWikiApiErrorException e) {
70        throw new RuntimeException(e);
71      }
72      if (entityDocument == null)
73        throw new IllegalArgumentException(
74            "Entity Document for " + entityId + " not found");
75      WikiDataNode node = new WikiDataNode(this, entityDocument, keys);
76      if (this.getStartNode() == null)
77        this.setStartNode(node);
78      return node;
79    }

97    /**
98     * create a WikiDataSystem with a given fetcher
99     * 
100    * @param wbdf
101    * @param languageCodes
102    */
103   private WikiDataSystem(WikibaseDataFetcher wbdf, String... languageCodes) {
104     super.setName("WikiData");
105     super.setVersion("0.0.8");
106     this.wbdf = wbdf;
107     for (String languageCode : languageCodes) {
108       addLanguage(languageCode);
109     }
110   }

126   /**
127    * enable the use of a cache for the given purpose
128    * 
129    * @param cacheFile
130    * @param purpose
131    * @return
132    * @throws Exception
133    */
134   public Cache useCache(File cacheFile, String purpose) throws Exception {
135     // create a new WikiDataSystem based on the same fetcher
136     WikiDataSystem cachews = new WikiDataSystem(wbdf,
137         languages.toArray(new String[0]));
138     Cache cache = new Cache(cachews, cacheFile, purpose);
139     cacheMap.put(purpose, cache);
140     // is there propertyCacheFile?
141     if (cacheFile.exists()) {
142       cache.reinit();
143     }
144     return cache;
145   }

149   /**
150    * cache the entity with the given id
151    * 
152    * @param entityId
153    *          - Property or Item
154    * @param level
155    */
156   public SimpleNode cache(EntityIdValue entityId, boolean optional) {
157     SimpleNode node = null;
158     String purpose = entityId.getEntityType();
159     // do we have a cache for the given purpose?
160     if (cacheMap.containsKey(purpose)) {
161       Cache cache = cacheMap.get(purpose);
162       String wikidata_id=entityId.getId();
163       // try to find the
164       GraphTraversal<Vertex, Vertex> propVertex = cache.getSystem().g().V()
165           .has("wikidata_id", wikidata_id);
166       if (propVertex.hasNext()) {
167         node = propVertex.next().value("mysimplenode");
168       } else {
169         // no cache entry
170         // if the caching is non optional get the value
171         if (!optional) {
172           node = cache.getSystem().moveTo(entityId.getId());
173           if (debug)
174             LOGGER.log(Level.INFO, String.format("caching %s=%s",
175                 entityId.getIri(), node.getMap().get("label_en")));
176         }
177       }
178     }
179     return node;
180   }

182   /**
183    * create an EntityIdValue
184    * 
185    * @param id
186    *          - the id to create the entityId value for
187    * @return the EntityId Value
188    */
189   public static Optional<EntityIdValue> createEntityIdValue(String id) {
190     if (!id.matches("^[QP][0-9]+$")) {
191       return Optional.empty();
192     }
193     // https://wikidata.github.io/Wikidata-Toolkit/org/wikidata/wdtk/datamodel/implementation/PropertyIdValueImpl.html
194     // https://stackoverflow.com/a/2599471/1497139
195     try {
196       Constructor<? extends EntityIdValue> constructor;
197       Class<? extends EntityIdValue> clazz=ItemIdValueImpl.class;
198       if (id.startsWith("P")) {
199         clazz=PropertyIdValueImpl.class;
200       };
201       constructor=clazz.getDeclaredConstructor(String.class,String.class);
202       constructor.setAccessible(true);      
203       EntityIdValue idValue = constructor.newInstance(id, siteiri);
204       return Optional.of(idValue);
205     } catch (Throwable th) {
206       throw new RuntimeException(th);
207     }
208   }

210   /**
211    * get a cache result by id
212    * 
213    * @param id
214    *          - a property or item id
215    * @param optional
216    * @return the node
217    */
218   public Optional<SimpleNode> cache(String id, boolean optional) {
219     Optional<EntityIdValue> idValue = createEntityIdValue(id);
220     if (idValue.isPresent()) {
221       SimpleNode node = this.cache(idValue.get(), optional);
222       return Optional.of(node);
223     } else {
224       return Optional.empty();
225     }
226   }
