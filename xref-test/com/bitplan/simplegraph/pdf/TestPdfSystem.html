30   * test the PDF System

34  public class TestPdfSystem {
35  	public static boolean debug = false;
36  	protected static Logger LOGGER = Logger.getLogger("com.bitplan.simplegraph.pdf");
37  
38  	@Test
39  	public void testPDF() throws Exception {
40  		PdfSystem ps = new PdfSystem();
41  		ps.connect();
42  		SimpleNode pdfNode = ps.moveTo("http://eprints.nottingham.ac.uk/249/1/cajun.pdf");
43  		pdfNode.out("pages");
44  		//debug=true;
45  		if (debug) {
46  			ps.forAll(SimpleNode.printDebug);
47  			System.out.println(ps.getStartNode().g().V().hasLabel("pdf").next().property("NumberOfPages").value());
48  			ps.getStartNode().g().V().hasLabel("page")
49  					.forEachRemaining(page -> System.out.println(page.property("text").value()));
50  		}
51  	}
