119   public Graph getFirstGraph() {
120     // http://tinkerpop.apache.org/docs/3.1.8/tutorials/getting-started/
121     Graph graph = TinkerGraph.open(); // 1
122     Vertex marko = graph.addVertex(T.label, "person", T.id, 1, "name", "marko",
123         "age", 29); // 2
124     Vertex vadas = graph.addVertex(T.label, "person", T.id, 2, "name", "vadas",
125         "age", 27);
126     Vertex lop = graph.addVertex(T.label, "software", T.id, 3, "name", "lop",
127         "lang", "java");
128     Vertex josh = graph.addVertex(T.label, "person", T.id, 4, "name", "josh",
129         "age", 32);
130     Vertex ripple = graph.addVertex(T.label, "software", T.id, 5, "name",
131         "ripple", "lang", "java");
132     Vertex peter = graph.addVertex(T.label, "person", T.id, 6, "name", "peter",
133         "age", 35);
134     Edge e = marko.addEdge("knows", vadas, T.id, 7, "weight", 0.5f); // 3
135     assertNotNull(e);
136     marko.addEdge("knows", josh, T.id, 8, "weight", 1.0f);
137     marko.addEdge("created", lop, T.id, 9, "weight", 0.4f);
138     josh.addEdge("created", ripple, T.id, 10, "weight", 1.0f);
139     josh.addEdge("created", lop, T.id, 11, "weight", 0.4f);
140     peter.addEdge("created", lop, T.id, 12, "weight", 0.2f);
141     return graph;
142   }

144   /**
145    * dump the given graph
146    * 
147    * @param graph
148    */
149   public void dumpGraph(Graph graph) {
150     if (debug) {
151       graph.traversal().V().forEachRemaining(
152           vertex -> System.out.println(String.format("%s=%s %s", vertex.id(),
153               vertex.label(), vertex.property("name").value())));
154       graph.traversal().E().forEachRemaining(
155           edge -> System.out.println(String.format("%s- %s >%s",
156               edge.inVertex().id(), edge.label(), edge.outVertex().id())));
157     }
158   }

185   @Test
186   public void testTutorial() {
187     // see http://tinkerpop.apache.org/docs/3.3.1/tutorials/getting-started/
188     Graph graph = TinkerFactory.createModern();
189     assertEquals(28.0,
190         graph.traversal().V().has("name", P.within("vadas", "marko"))
191             .values("age").mean().next().doubleValue(),
192         0.001);
193     List<Object> creators = graph.traversal().V().has("name", "marko")
194         .out("created").in("created").values("name").toStream()
195         .collect(Collectors.toList());
196     assertEquals(3, creators.size());
197     String creatorNames = String.join(",",
198         creators.toArray(new String[creators.size()]));
199     assertEquals("marko,josh,peter", creatorNames);
200 
201     List<Object> creatorsWithOutMarko = graph.traversal().V()
202         .has("name", "marko").as("exclude").out("created").in("created")
203         .where(P.neq("exclude")).values("name").toStream()
204         .collect(Collectors.toList());
205     assertEquals(2, creatorsWithOutMarko.size());
206     String creatorNamesWithOutMarko = String.join(",",
207         creatorsWithOutMarko.toArray(new String[creatorsWithOutMarko.size()]));
208     assertEquals("josh,peter", creatorNamesWithOutMarko);
209 
210     List<Map<String, Object>> selected = graph.traversal().V().as("a").out()
211         .as("b").out().as("c").select("a", "b", "c").toStream()
212         .collect(Collectors.toList());
213     assertEquals(2, selected.size());
214     if (debug)
215       selected.forEach(SimpleNode.printMapDebug);
216     List<Map<Object, Object>> groups = graph.traversal().V().group().by(T.label)
217         .toStream().collect(Collectors.toList());
218     if (debug)
219       groups.forEach(SimpleNode.printMapDebug);
220     assertEquals(1, groups.size());
221     // debug = true;
222     List<Map<Object, Object>> groupsByName = graph.traversal().V().group()
223         .by(T.label).by("name").toStream().collect(Collectors.toList());
224     if (debug)
225       groupsByName.forEach(SimpleNode.printMapDebug);
226   }

229   /**
230    * test according to https://github.com/krlawrence/graph
231    * http://kelvinlawrence.net/book/Gremlin-Graph-Guide.html
232    * http://kelvinlawrence.net/book/Gremlin-Graph-Guide.pdf

234   public void testGuide() throws Exception {
235     Graph graph = getAirRoutes();
236     GraphTraversalSource g = graph.traversal();
237     // 3.1.1. A quick look at Gremlin and SQL
238     // g.V().hasLabel('airport').groupCount().by('country')
239     Map<Object, Long> groupCount = g.V().hasLabel("airport").groupCount()
240         .by("country").next();
241     if (debug)
242       groupCount.entrySet().forEach(entry -> System.out
243           .println(String.format("%s=%s", entry.getKey(), entry.getValue())));
244     // uncomment if you"d like to see all available ids
245     // g.V().forEachRemaining(v->System.out.println(v.id()));
246     // Query the properties of vertex 3
247     GraphTraversal<Vertex, Object> unfolded1 = g.V("3").valueMap(true).unfold();
248     // debug = true;
249     if (debug) {
250       unfolded1.forEachRemaining(o -> System.out.println(o.toString()));
251       /**
252        * country=[US] code=[AUS] longest=[12250] city=[Austin] id=3
253        * lon=[-97.6698989868164] type=[airport] label=airport elev=[542]
254        * icao=[KAUS] region=[US-TX] runways=[2] lat=[30.1944999694824]
255        * desc=[Austin Bergstrom International Airport]
256        */
257     }
258     assertEquals(14,
259         g.V("3").valueMap(true).unfold().count().next().longValue());
260     GraphTraversal<Vertex, Vertex> v3 = g.V(3);
261     GraphTraversal<Vertex, Map<Object, Object>> vmap = v3.valueMap(true);
262     GraphTraversal<Vertex, Object> unfolded = g.V(3).valueMap(true).unfold();
263     if (debug)
264       unfolded.forEachRemaining(o -> System.out.println(
265           String.format("%s (%s)", o.toString(), o.getClass().getName())));
266     if (debug)
267       g.V().hasLabel("airport").values("code")
268           .forEachRemaining(o -> System.out.println(o.toString()));
269 
270     // g.V().has('code','AUS').out().out().out().has('code','AGR').path().by('code')
271     GraphTraversal<Vertex, Path> paths = g.V().has("code", "AUS").out().out()
272         .out().has("code", "AGR").path().by("code");
273     assertEquals(4, paths.count().next().longValue());
274     paths = g.V().has("code", "AUS").out().out().out().has("code", "AGR").path()
275         .by("code");
276     if (debug)
277       paths.forEachRemaining(path -> {
278         System.out.println(path.toString());
279       });
280     // g.V().has('code','AUS').repeat(out()).times(3).has('code','AGR').path().by('code')
281 
282     if (debug)
283       g.V().has("code", "AUS").repeat(__.out()).times(3).has("code", "AGR")
284           .path().by("code").forEachRemaining(path -> {
285             System.out.println(path.toString());
286           });
287     // fail("tschüss");
288     // Find vertices that are airports
289     // g.V().hasLabel('airport')
290     assertEquals(3374, g.V().hasLabel("airport").count().next().longValue());
291     // Find the DFW vertex
292     // g.V().has('code','DFW')
293     assertEquals("Dallas",
294         g.V().has("code", "DFW").next().property("city").value());
295     // Combining those two previous queries (two ways that are equivalent)
296     // g.V().hasLabel('airport').has('code','DFW')
297     assertEquals("8", g.V().hasLabel("airport").has("code", "DFW").next().id());
298     // g.V().has('airport','code','DFW')
299     assertEquals("8", g.V().has("airport", "code", "DFW").next().id());
300     // g.V().has('airport','code','DFW').next().getClass()
301     assertEquals("TinkerVertex",
302         g.V().has("airport", "code", "DFW").next().getClass().getSimpleName());
303     // 3.2.1. Retrieving property values from a vertex
304     // What property values are stored in the DFW vertex?
305     // g.V().has('airport','code','DFW').values()
306     List<Object> values = g.V().has("airport", "code", "DFW").values().toList();
307     assertEquals(12, values.size());
308     // debug=true;
309     if (debug)
310       values.forEach(value -> System.out.println(value));
311     // Return just the city name property
312     // fail("tschüss");
313     // g.V().has('airport','code','DFW').values('city')
314     assertEquals("Dallas", g.V().has("code", "DFW").values("city").next());
315     // Return the 'runways' and 'icao' property values.
316     // g.V().has('airport','code','DFW').values('runways','icao')
317     GraphTraversal<Vertex, Object> rit = g.V().has("code", "DFW")
318         .values("runways", "icao");
319     assertEquals("KDFW", rit.next());
320     assertEquals(7, rit.next());
321     // Does a specific property exist on a given vertex or edge?
322 
323     // Find all edges that have a 'dist' property
324     // g.E().has('dist')
325     g.E().has("dist").forEachRemaining(e -> {
326       if (debug) {
327         System.out.println(String.format("%s -> %s %3d miles",
328             e.inVertex().values("city").next(),
329             e.outVertex().values("city").next(), e.property("dist").value()));
330       }
331     });
332 
333     // Find all vertices that have a 'region' property
334     // g.V().has('region')
335     assertEquals(3374, g.V().has("region").count().next().longValue());
336 
337     // Find all the vertices that do not have a 'region' property
338     // g.V().hasNot('region')
339     assertEquals(245, g.V().hasNot("region").count().next().longValue());
340 
341     // The above is shorthand for
342     // g.V().not(has('region'))
343     assertEquals(245, g.V().not(__.has("region")).count().next().longValue());
344     // 3.2.3. Counting things
345     // How many airports are there in the graph?
346     // g.V().hasLabel('airport').count()
347     assertEquals(3374, g.V().hasLabel("airport").count().next().longValue());
348     // How many routes are there?
349     // g.V().hasLabel('airport').outE('route').count()
350     assertEquals(43400,
351         g.V().hasLabel("airport").outE("route").count().next().longValue());
352     // How many routes are there?
353     // g.V().outE('route').count()
354     assertEquals(43400, g.V().outE("route").count().next().longValue());
355     // // How many routes are there?
356     // g.E().hasLabel('route').count()
357     assertEquals(43400, g.E().hasLabel("route").count().next().longValue());
358   }

402       Edge edge = g.E().hasLabel(edgeName).next();
403       if (debug)
404         System.out.println(String.format("%s -> %s: %s",
405             edge.outVertex().label(), edge.inVertex().label(), edge.label()));
406     }
