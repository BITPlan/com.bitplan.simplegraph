70    /**
71     * 
72     * @param sls
73     * @throws Exception
74     */
75    public Slide slideForNode(SlideShow sls, SimpleNode node, String... props)
76        throws Exception {
77      SlideNode slide = (SlideNode) sls.createSlide();
78      Map<String, Object> map = node.getMap();
79      String source = map.get("source").toString();
80      slide.property("source", source);
81      int y = 20;
82      int x = 235;
83      int lwidth = 115;
84      int twidth = 350;
85      double fontSize = 16.0;
86      int height = (int) (fontSize * 1.5);
87      int imageheight = 800;
88      String name = map.get("name").toString();
89      slide.setTitle(name);
90      String url = null;
91      if ("WikiData".equals(source))
92        url = "https://tools.wmflabs.org/sqid/#/view?id="
93            + node.getProperty("wikidata_id");
94      else
95        url = "http://royal-family.bitplan.com/index.php/"
96            + name.replaceAll(" ", "_");
97      slide.addText(slide.addTextBox(x, y, lwidth, height), "link:", fontSize,
98          Color.black);
99      slide.addHyperlink(slide.addTextBox(x + lwidth, y, twidth, height),
100         map.get("name").toString(), fontSize, Color.blue, url);
101     y += height;
102     for (String prop : props) {
103       Object propObj = node.getProperty(prop);
104       if (propObj != null) {
105         if (propObj instanceof String) {
106           String text = propObj.toString();
107           slide.addText(slide.addTextBox(x, y, lwidth, height), prop + ":",
108               fontSize, Color.black);
109           double tFontSize = fontSize;
110           if (text.length() > 45)
111             tFontSize = fontSize * 0.75;
112           slide.addText(slide.addTextBox(x + lwidth, y, twidth, height),
113               propObj.toString(), tFontSize, Color.blue);
114           y += height;
115           // if (y > imageheight) {
116           // x = 20;
117           // }
118         } else if (propObj instanceof BufferedImage) {
119           BufferedImage image = (BufferedImage) propObj;
120           XSLFPictureShape picture = slide.addPicture(image);
121           if (picture != null) {
122             imageheight = image.getHeight();
123             slide.setPosition(picture, 20, y - height, image.getWidth(),
124                 imageheight);
125           }
126         } else {
127           LOGGER.log(Level.INFO, String.format("property %s=(%s) %s", prop,
128               propObj.getClass().getSimpleName(), propObj.toString()));
129         }
130 
131       }
132     }
133     ByteArrayOutputStream baos = new ByteArrayOutputStream();
134     PrintStream ps = new PrintStream(baos, true, "utf-8");
135     node.printNameValues(ps);
136     ps.close();
137     slide.setNotes(new String(baos.toByteArray(), StandardCharsets.UTF_8));
138     return slide;
139   }

141   /**
142    * get a new powerpoint file for the given path and title
143    * 
144    * @param pptFilePath
145    * @param title
146    * @return
147    * @throws Exception
148    */
149   public SlideShow getPPT(String pptFilePath, String title) throws Exception {
150     PowerPointSystem pps = new PowerPointSystem();
151     File pptFile = new File(pptFilePath);
152     // remove file if it exists
153     if (pptFile.exists())
154       pptFile.delete();
155     // to force to create a new empty slide show
156     SlideShow sls = (SlideShow) pps.connect("").moveTo(pptFilePath);
157     sls.setTitle(title);
158     return sls;
159   }

161   /**
162    * get the image for the given person
163    * 
164    * @param mws
165    * @param personNode
166    * @return - the buffered image
167    * @throws Exception
168    */
169   public BufferedImage getImage(MediaWikiSystem mws, SimpleNode personNode,
170       int size) throws Exception {
171     if (personNode==null)
172       return null;
173     String image = personNode.getProperty("image").toString();
174     MediaWikiPageNode pageNode = (MediaWikiPageNode) mws
175         .moveTo("File:" + image);
176     return pageNode.getImage(size);
177   }

179   /**
180    * add a slide for the given person
181    * 
182    * @param sls
183    * @param mws
184    * @param person
185    * @param slideprops
186    * @param debug
187    * @throws Exception
188    */
189   private void addSlide(SlideShow sls, MediaWikiSystem mws, SimpleNode person,
190       String[] slideprops, boolean debug) throws Exception {
191     // add a property "source" to the node
192     String source = "WikiData";
193     if (person.getMap().containsKey("P18")) {
194       BufferedImage image = this.getImage(mws, person, 200);
195       if (image != null)
196         person.property("picture", image);
197     }
198     person.property("source", source);
199     person.property("name", person.getMap().get("label_en"));
200     SlideNode personNode = (SlideNode) slideForNode(sls, person, slideprops);
201     if (debug)
202       personNode.printNameValues(System.out);
203   }

205   /**
206    * add slides for children
207    * 
208    * @param sls
209    * @param mws
210    * @param mws
211    * @param person
212    * @param levels
213    * @throws Exception
214    */
215   private void addSlides(SlideShow sls, MediaWikiSystem mws, SimpleNode person,
216       String[] slideprops, int levels, boolean debug) throws Exception {
217     this.addSlide(sls, mws, person, slideprops, debug);
218     List<SimpleNode> children = person.out("child")
219         .collect(Collectors.toCollection(ArrayList::new));
220     for (SimpleNode child : children) {
221       if (levels >= 1) {
222         addSlides(sls, mws, child, slideprops, levels - 1, debug);
223       }
224     }
225   }

251   @Test
252   public void testPowerPointCreateRoyal92() throws Exception {
253     SimpleSystem royal92 = TestTripleStoreSystem.readSiDIF();
254     // start with Queen Victoria (Person Id=I1)
255     SimpleNode queenVictoria = royal92.moveTo("id=I1");
256     String pptFilePath = "../simplegraph-powerpoint/QueenVictoriaRoyal92.pptx";
257     SlideShow sls = this.getPPT(pptFilePath, "Queen Victoria");
258 
259     /**
260      * born = 1819-05-24 sex = female nobleTitle = Queen of England childOf =
261      * F42 yearBorn = 1819 parentOf = F1 died = 1901-01-22 diedAt = Royal
262      * Mausoleum,Frogmore,Berkshire,England birthPlace =
263      * Kensington,Palace,London,England yearDied = 1901 isA = Person monthBorn =
264      * 5 name = Victoria Hanover id = I1 monthDied = 1
265      */
266     String slideprops[] = { "name", "nobleTitle", "sex", "yearBorn",
267         "birthPlace", "yearDied", "diedAt", "source" };
268     // add slides
269     String source = "Royal 92 GEDCOM file";
270     queenVictoria.property("source", source);
271     SlideNode qv = (SlideNode) slideForNode(sls, queenVictoria, slideprops);
272     assertNotNull(qv);
273     // debug = true;
274     for (SimpleNode child : TestTripleStoreSystem.children(queenVictoria, 2)) {
275       child.property("source", source);
276       SlideNode slide = (SlideNode) slideForNode(sls, child, slideprops);
277       assertEquals(slide.getTitle(), child.getMap().get("name"));
278       String text = slide.getText(SlideNode.SEPARATOR);
279       if (debug)
280         System.out.println(text);
281       assertTrue(text.contains(child.getMap().get("nobleTitle").toString()));
282       String notes = slide.getNotes(SlideNode.SEPARATOR);
283       if (debug)
284         System.out.println(notes);
285       // check that the newly set property value source is available wher expect
286       String csource = slide.getMap().get("source").toString();
287       assertEquals(source, csource);
288       assertEquals(source,
289           slide.getVertex().property("source").value().toString());
290       assertTrue(notes.contains(child.getMap().get("childOf").toString()));
291       assertTrue(notes.contains(source));
292     }
293     sls.save();
294   }
